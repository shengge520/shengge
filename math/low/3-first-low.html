# 第三题：跳台阶/斐波那契

function ttj(n){
    if(n<0) throw new Error('输入的数字不能小于0');
    if(n==1 || n==2){
        return 1;
    } else {
        return ttj(n-1) + ttj(n-2);
    }

}
